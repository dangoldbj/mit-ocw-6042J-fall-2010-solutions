    \item $\exists x \in X$ such that $S(x) \land A(x)$ is true.
    \item $\forall x \in X, T(x) \land S(x) \implies A(x)$
    \item $\nexists x \in X$ such that $(T(x) \land S(x)) \land \neg A(x)$
    \item $S = \{ \forall x \in X : T(x) \land \neg S(x) \}$. Then, $|S| \geq 3$
}
\end{solution}
